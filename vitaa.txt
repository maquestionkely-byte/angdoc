<script>
    (function() {
        // --------------------------------------------------------------
        // 1. SORATRA MIADANA BE (lettre par lettre ~210ms) → tena mampionona
        // --------------------------------------------------------------
        const fullPeaceText = "🌿 Asio tsiky kely oe jereko ! Manisa miadana : 5 ... 3 ... 2 ... 1 ... 0 ... mitsiky malefaka. Alefaso ny tsiky, mba tonono koa hoe soasisy manify tsy nisy sesiny de nanify, aia solonify. Mifalia moramora, avelao ny tsiky hitombo... 🌿";
        
        const typewriterDiv = document.getElementById("typewriterMessage");
        let currentIdx = 0;
        let typingInterval = null;
        
        function startVerySlowTyping() {
            if (typingInterval) clearInterval(typingInterval);
            typewriterDiv.innerHTML = "";
            currentIdx = 0;
            // 210 milliseconde par lettre -> extrêmement lent, idéal pour la relaxation
            typingInterval = setInterval(() => {
                if (currentIdx < fullPeaceText.length) {
                    let char = fullPeaceText[currentIdx];
                    typewriterDiv.innerHTML += char;
                    currentIdx++;
                    if (currentIdx === fullPeaceText.length) {
                        clearInterval(typingInterval);
                        typingInterval = null;
                        let cursorSpan = document.createElement("span");
                        cursorSpan.className = "cursor-blink";
                        cursorSpan.innerHTML = " ";
                        typewriterDiv.appendChild(cursorSpan);
                    }
                } else {
                    clearInterval(typingInterval);
                    typingInterval = null;
                }
            }, 210); // 210ms → très lent, chaque lettre apparaît doucement
        }
        
        startVerySlowTyping();
        
        // --------------------------------------------------------------
        // 2. MINUTEUR LENT : chaque chiffre dure 2 secondes
        //    Séquence : "5 triste" , "3 😍" , "2 🙂" , "1 🙂" , "0 rire léger"
        //    Cycle complet = 5 chiffres × 2 secondes = 10 secondes
        //    Boucle automatique infinie
        // --------------------------------------------------------------
        
        const countdownNumberSpan = document.getElementById("countdownNumber");
        const emotionFaceSpan = document.getElementById("emotionFace");
        const cycleInfoSpan = document.getElementById("cycleInfo");
        
        // Définition des étapes (chiffre + émotion + message)
        const steps = [
            { digit: "5", emotion: "🥰", label: "✨ 5 triste... avelao hidina ny alahelo ✨" },
            { digit: "3", emotion: "😍", label: "💖 3 fitiavana sy hafaliana 💖" },
            { digit: "2", emotion: "🙂", label: "🌼 2 tsiky kely malefaka 🌼" },
            { digit: "1", emotion: "🙂", label: "🍃 1 manakaiky ny fahafinaretana 🍃" },
            { digit: "0", emotion: "😆", label: "🎈 0 rire léger ! faly sy maivana 🎈" }
        ];
        
        let currentStepIndex = 0;
        let timerInterval = null;
        
        function updateDisplayForStep(stepIndex) {
            const step = steps[stepIndex];
            if (!step) return;
            // animation douce au changement
            countdownNumberSpan.style.transform = "scale(1.08)";
            emotionFaceSpan.style.transform = "scale(1.05)";
            countdownNumberSpan.innerText = step.digit;
            emotionFaceSpan.innerHTML = step.emotion;
            cycleInfoSpan.innerHTML = step.label + "  ~  " + (step.digit === "5" ? "2 segondra isa" : "2 segondra");
            
            // Effet de lumière supplémentaire pour le chiffre 0 (rire léger)
            if (step.digit === "0") {
                cycleInfoSpan.style.fontWeight = "bold";
                timerSection.style.borderColor = "#ffd966";
            } else {
                cycleInfoSpan.style.fontWeight = "normal";
                timerSection.style.borderColor = "rgba(215, 185, 110, 0.4)";
            }
            
            setTimeout(() => {
                if (countdownNumberSpan) countdownNumberSpan.style.transform = "scale(1)";
                if (emotionFaceSpan) emotionFaceSpan.style.transform = "scale(1)";
            }, 280);
        }
        
        function goToNextStep() {
            currentStepIndex++;
            if (currentStepIndex >= steps.length) {
                // Fiafaran'ny cycle: miverina manomboka amin'ny 5 triste
                currentStepIndex = 0;
                // fanamarihana maivana rehefa miverina
                cycleInfoSpan.style.opacity = "0.7";
                setTimeout(() => {
                    if (cycleInfoSpan) cycleInfoSpan.style.opacity = "1";
                }, 500);
            }
            updateDisplayForStep(currentStepIndex);
        }
        
        function startVerySlowCyclicTimer() {
            if (timerInterval) clearInterval(timerInterval);
            // Manomboka amin'ny voalohany (5 triste)
            currentStepIndex = 0;
            updateDisplayForStep(currentStepIndex);
            // Isaky ny 2000 ms (2 segondra) dia miova mankany amin'ny étape manaraka
            timerInterval = setInterval(() => {
                goToNextStep();
            }, 2000); // 2 secondes par chiffre → total 10 secondes le cycle
        }
        
        startVerySlowCyclicTimer();
        
        // Référence pour la section timer (effets visuels)
        const timerSection = document.querySelector('.timer-emotion-section');
        
        // Surveiller le passage au chiffre 0 pour un petit effet "rire léger" supplémentaire
        let previousDigit = null;
        setInterval(() => {
            const currentDigit = countdownNumberSpan.innerText;
            if (currentDigit === "0" && previousDigit !== "0") {
                // Animation zen quand on atteint 0
                if (timerSection) {
                    timerSection.style.transition = "background 0.5s, box-shadow 0.3s";
                    timerSection.style.background = "rgba(80, 105, 70, 0.5)";
                    timerSection.style.boxShadow = "0 0 12px #ffd58c";
                    setTimeout(() => {
                        if(timerSection) {
                            timerSection.style.background = "";
                            timerSection.style.boxShadow = "";
                        }
                    }, 700);
                }
                // Faire danser le smiley du 0 (rire)
                emotionFaceSpan.style.animation = "gentleFloat 0.8s ease-in-out";
                setTimeout(() => {
                    if(emotionFaceSpan) emotionFaceSpan.style.animation = "gentleFloat 6s infinite ease-in-out";
                }, 800);
            }
            previousDigit = currentDigit;
        }, 300);
        
        // Effet spécial pour le chiffre 5 (triste) : ambiance plus tamisée
        setInterval(() => {
            if (countdownNumberSpan.innerText === "5") {
                timerSection.style.borderLeft = "3px solid #9abf8f55";
                emotionFaceSpan.style.filter = "drop-shadow(0 2px 5px #5f7a5a)";
            } else {
                timerSection.style.borderLeft = "";
                emotionFaceSpan.style.filter = "";
            }
        }, 400);
        
        // Lorsque l'on est à 3 (😍), on ajoute une petite lueur chaleureuse
        setInterval(() => {
            if (countdownNumberSpan.innerText === "3") {
                emotionFaceSpan.style.textShadow = "0 0 12px #ffbc6e";
                countdownNumberSpan.style.textShadow = "0 0 18px #ffbc6e";
                setTimeout(() => {
                    if(emotionFaceSpan) emotionFaceSpan.style.textShadow = "";
                    if(countdownNumberSpan) countdownNumberSpan.style.textShadow = "0 0 14px #ffbf69";
                }, 500);
            }
        }, 500);
        
        // --------------------------------------------------------------
        // 3. AMBIANCE LENTE ET VARIATIONS DÉLICATES
        // --------------------------------------------------------------
        const calmCard = document.querySelector('.calm-card');
        let bgCycle = 0;
        setInterval(() => {
            if (calmCard) {
                bgCycle = (bgCycle + 1) % 3;
                const softColors = ["rgba(32, 52, 44, 0.6)", "rgba(42, 62, 50, 0.62)", "rgba(28, 48, 40, 0.58)"];
                calmCard.style.transition = "background 8s ease";
                calmCard.style.background = softColors[bgCycle % softColors.length];
            }
        }, 11000);
        
        // Générer quelques "lucioles" très lentes pour la sérénité
        function createSlowFirefly() {
            const firefly = document.createElement('div');
            firefly.style.position = 'fixed';
            firefly.style.width = '7px';
            firefly.style.height = '7px';
            firefly.style.background = 'radial-gradient(circle, #ffefb0, #ffbc6e)';
            firefly.style.borderRadius = '50%';
            firefly.style.pointerEvents = 'none';
            firefly.style.zIndex = '3';
            firefly.style.opacity = '0.35';
            firefly.style.filter = 'blur(2px)';
            firefly.style.left = Math.random() * 95 + '%';
            firefly.style.top = Math.random() * 95 + '%';
            document.body.appendChild(firefly);
            setTimeout(() => {
                firefly.style.transition = 'all 9s ease';
                firefly.style.opacity = '0';
                firefly.style.transform = 'translateY(-40px) translateX(15px)';
                setTimeout(() => firefly.remove(), 10000);
            }, 1200);
        }
        
        setInterval(() => {
            if (Math.random() > 0.7) createSlowFirefly();
        }, 9000);
        
        // Message de cycle qui s'adapte (pour souligner la lenteur)
        setInterval(() => {
            const currentDigit = countdownNumberSpan.innerText;
            if (currentDigit === "0") {
                cycleInfoSpan.style.transition = "color 0.5s";
                cycleInfoSpan.style.color = "#ffeaac";
                setTimeout(() => { if(cycleInfoSpan) cycleInfoSpan.style.color = "#daf3b3"; }, 700);
            } else if (currentDigit === "5") {
                cycleInfoSpan.style.color = "#cfe5bc";
            } else {
                cycleInfoSpan.style.color = "#daf3b3";
            }
        }, 1000);
        
        // Animation supplémentaire : le grand chiffre respire très lentement
        setInterval(() => {
            if (countdownNumberSpan) {
                countdownNumberSpan.style.transform = "scale(1.02)";
                setTimeout(() => {
                    if(countdownNumberSpan) countdownNumberSpan.style.transform = "scale(1)";
                }, 1200);
            }
        }, 3800);
        
        // --------------------------------------------------------------
        // 4. CONTRÔLE DE LA LENTEUR : bien vérifier que chaque chiffre dure 2 secondes
        //    et cycle complet de 10 secondes. Aucune action requise, tout est automatique.
        //    Les émotions suivent la demande : 5 triste, 3 😍, 2 🙂, 1 🙂, 0 rire léger.
        //    Le texte principal s'écrit aussi très lentement (210ms par lettre).
        // --------------------------------------------------------------
        
        // Afficher une note dans la console pour confirmer la lenteur (effet relax)
        console.log("🌿 Minuteur très lent : 2 secondes par chiffre → 10 secondes par cycle. Écriture à 210ms/lettre. Apaisement total.");
        
        // Petite touche: faire bouger les feuilles encore plus lentement
        const leaves = document.querySelector('.floating-leaves');
        if (leaves) {
            leaves.style.animationDuration = "36s";
        }
        
        // Pour le "rire léger" à 0, ajouter une micro variation de l'icône
        let lastZero = false;
        setInterval(() => {
            if (countdownNumberSpan.innerText === "0" && !lastZero) {
                emotionFaceSpan.style.transform = "rotate(2deg) scale(1.03)";
                setTimeout(() => {
                    if(emotionFaceSpan) emotionFaceSpan.style.transform = "";
                }, 800);
                lastZero = true;
            } else if (countdownNumberSpan.innerText !== "0") {
                lastZero = false;
            }
        }, 400);
        
        // Rendre le curseur encore plus paisible
        const style = document.createElement('style');
        style.textContent = `::selection { background: #c5b87a55; }`;
        document.head.appendChild(style);
        
        // Eviter tout clic perturbateur (mais garder l'expérience fluide)
        document.querySelectorAll('.big-digit, .emotion-emoji').forEach(el => {
            el.style.cursor = "default";
        });
        
        // Si jamais l'utilisateur quitte la page puis revient, le minuteur continue normalement (pas de souci)
        // On s'assure que le timer ne se dérègle pas à cause de la visibilité
        document.addEventListener("visibilitychange", () => {
            if (!document.hidden) {
                // Rafraîchir l'affichage sans casser le timer
                if (countdownNumberSpan) {
                    // juste une petite mise à jour mentale, le timer continue
                }
            }
        });
    })();
