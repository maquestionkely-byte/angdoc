<script>
// ==================== CONFIGURATION ====================
const WORKER_URL = 'https://passage.jeananglais8.workers.dev';
const FICHIER = 'pourpc5.txt';
const DB_NAME = 'lecoaan_pourpc5';
const DB_VERSION = 1;
const DATE_INFO_URL = WORKER_URL + '/date_rehetra.txt';

// 🔒 Utilisé par le worker mais masqué à l'écran
const NIVEAU = 'pcrehetra';

let deviceModel = localStorage.getItem('deviceModel');
if (!deviceModel) {
    deviceModel = 'Device-' + Math.random().toString(36).substr(2, 10);
    localStorage.setItem('deviceModel', deviceModel);
}
console.log('🔑 Device Model:', deviceModel);

let pendingRequest = null;
let contenuLecon = null;

// ==================== INDEXEDDB ====================
function ouvrirDB() {
    return new Promise((resolve, reject) => {
        const request = indexedDB.open(DB_NAME, DB_VERSION);
        request.onerror = () => reject(request.error);
        request.onsuccess = (e) => resolve(e.target.result);
        request.onupgradeneeded = (e) => {
            const db = e.target.result;
            if (!db.objectStoreNames.contains('contenu')) db.createObjectStore('contenu');
            if (!db.objectStoreNames.contains('version')) db.createObjectStore('version');
        };
    });
}

async function stockerContenu(contenu, version) {
    const db = await ouvrirDB();
    return new Promise((resolve, reject) => {
        const tx = db.transaction(['contenu', 'version'], 'readwrite');
        tx.objectStore('contenu').put(contenu, 'htmlContent');
        tx.objectStore('version').put(version, 'versionDate');
        tx.oncomplete = () => resolve();
        tx.onerror = () => reject(tx.error);
    });
}

async function recupererContenuCache() {
    try {
        const db = await ouvrirDB();
        return new Promise((resolve, reject) => {
            const tx = db.transaction(['contenu'], 'readonly');
            const req = tx.objectStore('contenu').get('htmlContent');
            req.onsuccess = () => resolve(req.result);
            req.onerror = () => reject(req.error);
        });
    } catch { return null; }
}

async function recupererVersionCache() {
    try {
        const db = await ouvrirDB();
        return new Promise((resolve, reject) => {
            const tx = db.transaction(['version'], 'readonly');
            const req = tx.objectStore('version').get('versionDate');
            req.onsuccess = () => resolve(req.result);
            req.onerror = () => reject(req.error);
        });
    } catch { return null; }
}

// ==================== RÉCUPÉRATION DE LA VERSION DISTANTE ====================
async function getRemoteVersion() {
    try {
        const response = await fetch(DATE_INFO_URL, { cache: 'no-cache' });
        if (!response.ok) return null;
        const content = await response.text();
        const blocks = content.split(/DERNIÈRE MISE À JOUR/);
        
        for (const block of blocks) {
            const fileMatch = block.match(/📁 Fichier:\s*([^\n]+)/);
            if (fileMatch && fileMatch[1].trim() === FICHIER) {
                const dayMatch = block.match(/📅\s*([\d\/]+)/);
                const timeMatch = block.match(/⏰\s*([\d:]+)/);
                if (dayMatch && timeMatch) {
                    return `${dayMatch[1].trim()} ${timeMatch[1].trim()}`;
                }
                const simpleMatch = block.match(/📆\s*([^\n]+)/);
                if (simpleMatch) return simpleMatch[1].trim();
            }
        }
        return null;
    } catch (error) {
        console.warn("Impossible de récupérer la version distante:", error);
        return null;
    }
}

// ==================== NOTIFICATIONS ====================
function afficherNotification(message, type = 'error') {
    const notif = document.createElement('div');
    notif.className = `update-notification ${type === 'success' ? 'success' : ''}`;
    notif.innerHTML = message;
    document.body.appendChild(notif);
    setTimeout(() => notif.remove(), 5000);
}

// ==================== OUVERTURE DE LA LEÇON ====================
function ouvrirLecon() {
    if (!contenuLecon) {
        afficherNotification('⚠️ Aucune leçon disponible', 'error');
        return;
    }
    const blob = new Blob([contenuLecon], { type: 'text/html' });
    const blobUrl = URL.createObjectURL(blob);
    window.location.href = blobUrl;
}

// ==================== TÉLÉCHARGEMENT DE LA LEÇON ====================
async function telechargerLecon() {
    const loadingModal = document.getElementById('loadingModal');
    const loadingText = document.getElementById('loadingText');
    
    loadingModal.classList.remove('hidden');
    loadingText.textContent = 'Téléchargement...';
    
    try {
        const url = `${WORKER_URL}/${FICHIER}?deviceModel=${encodeURIComponent(deviceModel)}&t=${Date.now()}`;
        const response = await fetch(url, { cache: 'no-cache' });
        
        if (response.status === 403) {
            try {
                const errorData = await response.json();
                if (errorData.needsCode) {
                    loadingModal.classList.add('hidden');
                    
                    // 🔒 Utilisation de NIVEAU = 'pcrehetra' (masqué)
                    pendingRequest = {
                        fileName: FICHIER,
                        niveau: NIVEAU,  // ← 'pcrehetra'
                        deviceModel: deviceModel,
                        titre: 'pourpc5.txt',
                        url: url
                    };
                    
                    // Les valeurs sont déjà dans les spans hidden-data
                    document.getElementById('codeModal').classList.add('active');
                    document.getElementById('modalCode').value = '';
                    document.getElementById('modalCode').focus();
                    return;
                }
            } catch (e) {
                loadingModal.classList.add('hidden');
                pendingRequest = {
                    fileName: FICHIER,
                    niveau: NIVEAU,  // ← 'pcrehetra'
                    deviceModel: deviceModel,
                    titre: 'pourpc5.txt',
                    url: url
                };
                document.getElementById('codeModal').classList.add('active');
                return;
            }
        }
        
        if (!response.ok) {
            throw new Error(`HTTP ${response.status}`);
        }
        
        const contenu = await response.text();
        if (!contenu || contenu.length < 50) {
            throw new Error('Contenu invalide');
        }
        
        const version = new Date().toLocaleString('fr-FR');
        await stockerContenu(contenu, version);
        contenuLecon = contenu;
        
        loadingModal.classList.add('hidden');
        ouvrirLecon();
        
    } catch (error) {
        loadingModal.classList.add('hidden');
        afficherNotification('⚠️ Erreur: ' + error.message, 'error');
    }
}

// ==================== VÉRIFICATION AVEC DATE_REHETRA ====================
async function verifierEtCharger() {
    const loadingModal = document.getElementById('loadingModal');
    const loadingText = document.getElementById('loadingText');
    
    loadingModal.classList.remove('hidden');
    loadingText.textContent = 'Vérification...';
    
    try {
        const versionLocale = await recupererVersionCache();
        const contenuCache = await recupererContenuCache();
        const versionDistante = await getRemoteVersion();
        
        if (contenuCache && versionLocale && versionDistante && versionLocale === versionDistante) {
            contenuLecon = contenuCache;
            loadingModal.classList.add('hidden');
            ouvrirLecon();
            return;
        }
        
        if (contenuCache && versionLocale && versionDistante && versionLocale !== versionDistante) {
            console.log('🔄 Mise à jour disponible, téléchargement...');
            await telechargerLecon();
            return;
        }
        
        if (!contenuCache || !versionDistante) {
            await telechargerLecon();
            return;
        }
        
        if (!versionDistante) {
            await telechargerLecon();
            return;
        }
        
    } catch (error) {
        console.error('❌ Erreur:', error);
        const contenuCache = await recupererContenuCache();
        if (contenuCache) {
            contenuLecon = contenuCache;
            loadingModal.classList.add('hidden');
            ouvrirLecon();
        } else {
            loadingModal.classList.add('hidden');
            afficherNotification('⚠️ Erreur: ' + error.message, 'error');
        }
    }
}

// ==================== SOUMISSION DU CODE ====================
async function submitModalCode() {
    const code = document.getElementById('modalCode').value.trim();
    if (!code) {
        afficherNotification('⚠️ Ampidiro ny code', 'error');
        return;
    }
    
    closeCodeModal();
    
    if (!pendingRequest) {
        afficherNotification('⚠️ Erreur: aucune requête en attente', 'error');
        return;
    }
    
    const loadingModal = document.getElementById('loadingModal');
    const loadingText = document.getElementById('loadingText');
    loadingModal.classList.remove('hidden');
    loadingText.textContent = 'Vérification du code...';
    
    try {
        const { fileName, niveau, deviceModel, url } = pendingRequest;
        
        const authResponse = await fetch(`${WORKER_URL}/api/authorize`, {
            method: 'POST',
            headers: { 'Content-Type': 'application/json' },
            body: JSON.stringify({ code, deviceModel, niveau })
        });
        
        const authData = await authResponse.json();
        
        if (authData.success) {
            loadingText.textContent = 'Téléchargement...';
            
            const fileResponse = await fetch(url);
            
            if (fileResponse.ok) {
                const contenu = await fileResponse.text();
                const version = new Date().toLocaleString('fr-FR');
                await stockerContenu(contenu, version);
                contenuLecon = contenu;
                
                loadingModal.classList.add('hidden');
                afficherNotification('✅ Code validé', 'success');
                ouvrirLecon();
            } else {
                throw new Error('Impossible de récupérer le fichier');
            }
        } else {
            loadingModal.classList.add('hidden');
            afficherNotification('❌ Code tsy mety!', 'error');
            
            document.getElementById('codeModal').classList.add('active');
            document.getElementById('modalCode').value = '';
            document.getElementById('modalCode').focus();
        }
    } catch (error) {
        loadingModal.classList.add('hidden');
        afficherNotification('❌ Erreur: ' + error.message, 'error');
        document.getElementById('codeModal').classList.add('active');
    }
}
