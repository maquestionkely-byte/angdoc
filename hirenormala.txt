        // ==================== CONFIGURATION ====================
        const FILES_CONFIG = {

            }

        const DB_NAME = 'irae_cache';
        const STORE_NAME = 'irae_contenus';
        
        let currentContent = null;

        let progressText = document.getElementById('progressText');
        let currentFileId = null;
        let currentFileUrl = null;
        let currentVersionUrl = null;
        
        // ==================== DÉTECTION DE L'APPAREIL ====================
        function detectDeviceSilent() {
            const ua = navigator.userAgent;
            let deviceType = "Inconnu";
            let os = "Inconnu";
            
            if (/Android/.test(ua)) {
                os = "Android";
                if (!/Mobile/.test(ua) || /Tablet/.test(ua)) {
                    deviceType = "Tablette Android";
                } else if (/Mobile/.test(ua)) {
                    deviceType = "Smartphone Android";
                } else {
                    deviceType = "Tablette Android";
                }
            }
            else if (/iPad/.test(ua) || (/Macintosh/.test(ua) && /Safari/.test(ua) && navigator.maxTouchPoints > 1)) {
                deviceType = "iPad";
                os = "iPadOS";
            }
            else if (/iPhone/.test(ua)) {
                deviceType = "iPhone";
                os = "iOS";
            }
            else if (/iPod/.test(ua)) {
                deviceType = "iPod";
                os = "iOS";
            }
            else if (/Windows/.test(ua)) {
                os = "Windows";
                deviceType = "PC Windows";
            }
            else if (/Macintosh/.test(ua)) {
                deviceType = "Mac";
                os = "macOS";
            }
            else if (/Linux/.test(ua) && !/Android/.test(ua)) {
                deviceType = "Linux";
                os = "Linux";
            }
            else if (/CrOS/.test(ua)) {
                deviceType = "Chromebook";
                os = "ChromeOS";
            }
            else if (/Mobile/.test(ua) || /Tablet/.test(ua)) {
                deviceType = "Mobile/Tablette";
                os = "Inconnu";
            }
            else {
                deviceType = "PC/Ordinateur";
                os = "Inconnu";
            }
            
            // Choix du fichier
            if (deviceType === "PC Windows" || deviceType === "Mac" || 
                deviceType === "Linux" || deviceType === "Chromebook" || 
                deviceType === "PC/Ordinateur") {
                currentFileId = FILES_CONFIG.pc.id;
                currentFileUrl = FILES_CONFIG.pc.url;
                currentVersionUrl = FILES_CONFIG.pc.versionUrl;
                console.log("🖥️ Détecté: PC ->", currentFileId);
            } else {
                currentFileId = FILES_CONFIG.mobile.id;
                currentFileUrl = FILES_CONFIG.mobile.url;
                currentVersionUrl = FILES_CONFIG.mobile.versionUrl;
                console.log("📱 Détecté: Mobile/Tablette ->", currentFileId);
            }
            
            return { deviceType, os, fileId: currentFileId, fileUrl: currentFileUrl };
        }
        
        // ==================== INDEXEDDB ====================
        function ouvrirDB() {
            return new Promise((resolve, reject) => {
                const request = indexedDB.open(DB_NAME, 2);
                request.onerror = () => reject(request.error);
                request.onsuccess = (event) => resolve(event.target.result);
                request.onupgradeneeded = (event) => {
                    const db = event.target.result;
                    if (!db.objectStoreNames.contains(STORE_NAME)) {
                        db.createObjectStore(STORE_NAME);
                        console.log("✅ Base IndexedDB créée");
                    }
                };
            });
        }
        
        async function sauvegarderDansCache(contenu, versionDate) {
            try {
                const db = await ouvrirDB();
                return new Promise((resolve, reject) => {
                    const transaction = db.transaction([STORE_NAME], 'readwrite');
                    const store = transaction.objectStore(STORE_NAME);
                    const request1 = store.put(contenu, currentFileId);
                    const request2 = store.put(versionDate, currentFileId + '_version');
                    
                    let completed = 0;
                    function checkComplete() {
                        completed++;
                        if (completed === 2) {
                            console.log("💾 Sauvegardé:", currentFileId);
                            resolve();
                        }
                    }
                    
                    request1.onsuccess = () => checkComplete();
                    request2.onsuccess = () => checkComplete();
                    request1.onerror = () => reject(request1.error);
                    request2.onerror = () => reject(request2.error);
                });
            } catch (error) {
                console.error("Erreur sauvegarde:", error);
                throw error;
            }
        }
        
        async function recupererDepuisCache() {
            try {
                const db = await ouvrirDB();
                return new Promise((resolve) => {
                    const transaction = db.transaction([STORE_NAME], 'readonly');
                    const store = transaction.objectStore(STORE_NAME);
                    const requestContent = store.get(currentFileId);
                    const requestVersion = store.get(currentFileId + '_version');
                    
                    let content = null;
                    let version = null;
                    let completed = 0;
                    
                    function checkComplete() {
                        completed++;
                        if (completed === 2) {
                            console.log("📀 Cache:", currentFileId, content ? "Trouvé" : "Vide");
                            resolve({ content, version });
                        }
                    }
                    
                    requestContent.onsuccess = () => { content = requestContent.result; checkComplete(); };
                    requestVersion.onsuccess = () => { version = requestVersion.result; checkComplete(); };
                    requestContent.onerror = () => checkComplete();
                    requestVersion.onerror = () => checkComplete();
                });
            } catch (error) {
                console.error("Erreur lecture cache:", error);
                return { content: null, version: null };
            }
        }
        
        // ==================== VÉRIFICATION DE MISE À JOUR ====================
        async function getRemoteVersionDate() {
            try {
                console.log("🌐 Vérification version distante pour:", currentFileId);
                
                const response = await fetch(currentVersionUrl, { 
                    cache: 'no-cache',
                    headers: { 'Cache-Control': 'no-cache' }
                });
                
                if (!response.ok) throw new Error(`HTTP ${response.status}`);
                const content = await response.text();
                
                // Déterminer le nom du fichier à chercher
                let searchFileName = '';
                if (currentFileId === 'rypc') {
                    searchFileName = 'hiremodepc.txt';
                } else {
                    searchFileName = 'hireola.txt';
                }
                
                // Chercher le bloc correspondant
                const blocks = content.split(/DERNIÈRE MISE À JOUR/);
                
                for (const block of blocks) {
                    const fileMatch = block.match(/📁 Fichier:\s*([^\n]+)/);
                    if (fileMatch && fileMatch[1].trim() === searchFileName) {
                        const dayMatch = block.match(/📅\s*([\d\/]+)/);
                        const timeMatch = block.match(/⏰\s*([\d:]+)/);
                        
                        if (dayMatch && timeMatch) {
                            const versionDate = `${dayMatch[1]} ${timeMatch[1]}`;
                            console.log("📅 Date trouvée pour", searchFileName, ":", versionDate);
                            return versionDate;
                        }
                    }
                }
                
                console.warn(`⚠️ Format non trouvé pour ${searchFileName}`);
                return null;
                
            } catch (error) {
                console.warn("Impossible de récupérer la version distante:", error);
                return null;
            }
        }
        
        // ==================== TÉLÉCHARGEMENT ====================
        async function telechargerEtAfficher(forceDownload = false) {
            if (progressText) progressText.innerText = '';
            
            try {
                // 1. Récupérer la version distante
                const remoteVersionDate = await getRemoteVersionDate();
                
                // 2. Si pas forcé, vérifier si version identique
                if (!forceDownload && currentVersionDate && remoteVersionDate && 
                    currentVersionDate === remoteVersionDate) {
                    console.log("✅ Version identique, pas de mise à jour");
                    if (progressText) progressText.innerText = 'Version à jour...';
                    return false;
                }
                
                console.log("🌐 Téléchargement depuis:", currentFileUrl);
                if (progressText) progressText.innerText = 'Téléchargement...';
                
                const response = await fetch(currentFileUrl, {
                    method: 'GET',
                    cache: 'no-store',
                    headers: { 'Accept': 'text/html, text/plain, */*' }
                });
                
                if (!response.ok) {
                    throw new Error(`HTTP ${response.status}: ${response.statusText}`);
                }
                
                if (progressText) progressText.innerText = 'Lecture du fichier...';
                const content = await response.text();
                
                if (!content || content.length < 100) {
                    throw new Error("Le fichier est vide ou trop court");
                }
                
                console.log("✅ Téléchargement réussi, taille:", content.length);
                
                const finalVersionDate = remoteVersionDate || new Date().toLocaleString('fr-FR');
                
                if (progressText) progressText.innerText = 'Stockage...';
                await sauvegarderDansCache(content, finalVersionDate);
                
                currentContent = content;
                currentVersionDate = finalVersionDate;
                
                if (progressText) progressText.innerText = 'Affichage...';
                afficherContenu(content);
                
                return true;
                
            } catch (error) {
                console.error("❌ Erreur:", error);
                throw error;
            }
        }
        
        // ==================== MISE À JOUR SILENCIEUSE ====================
        async function miseAJourSilencieuse() {
            try {
                console.log("🔍 Vérification mise à jour en arrière-plan...");
                const remoteVersionDate = await getRemoteVersionDate();
                
                if (!remoteVersionDate) {
                    console.log("Impossible de vérifier la version distante");
                    return;
                }
                
                if (currentVersionDate && currentVersionDate === remoteVersionDate) {
                    console.log("✅ Version à jour, pas de mise à jour nécessaire");
                    return;
                }
                
                console.log("🔄 Nouvelle version détectée, mise à jour...");
                await telechargerEtAfficher(true);
                
            } catch (error) {
                console.error("Erreur mise à jour silencieuse:", error);
            }
        }
        
        // ==================== AFFICHAGE ====================
        function afficherContenu(htmlContent) {
            if (loaderContainer) {
                loaderContainer.style.opacity = '0';
                setTimeout(() => {
                    loaderContainer.classList.add('hidden');
                }, 500);
            }
            
            document.body.innerHTML = '';
            document.body.style.margin = '0';
            document.body.style.padding = '0';
            
            const iframe = document.createElement('iframe');
            iframe.style.position = 'fixed';
            iframe.style.top = '0';
            iframe.style.left = '0';
            iframe.style.width = '100%';
            iframe.style.height = '100%';
            iframe.style.border = 'none';
            iframe.style.zIndex = '1';
            iframe.style.backgroundColor = 'white';
            iframe.srcdoc = htmlContent;
            document.body.appendChild(iframe);
            
            console.log("🎯 Contenu affiché pour:", currentFileId);
        }

        function afficherErreur(message) {
            if (loaderContainer) {
                loaderContainer.innerHTML = `
                    <div class="error-card">
                        <h2>⚠️ Petit problème</h2>
                        <p>${message}</p>
                        <button onclick="location.reload()">Réessayer</button>
                    </div>
                `;
                loaderContainer.style.background = 'linear-gradient(135deg, #667eea 0%, #764ba2 100%)';
            }
        }
        
        // ==================== SERVICE WORKER ====================
        async function registerServiceWorker() {
            if (!('serviceWorker' in navigator)) return;
            try {
                await navigator.serviceWorker.register('/sw.js');
                console.log('✅ Service Worker enregistré');
            } catch (error) {
                console.log('Service Worker non disponible');
            }
        }
        
        // ==================== CHARGEMENT AUTOMATIQUE ====================
        async function autoLoad() {
            try {
                // 1. Détection silencieuse
                const deviceInfo = detectDeviceSilent();
                console.log("📱 Appareil détecté:", deviceInfo);
                
                // 2. Vérifier le cache
                if (progressText) progressText.innerText = 'Vérification du cache...';
                const cached = await recupererDepuisCache();
                
                if (cached.content && cached.content.length > 100) {
                    // Cache trouvé - afficher immédiatement
                    console.log("✅ Cache trouvé, affichage immédiat");
                    currentContent = cached.content;
                    currentVersionDate = cached.version;
                    afficherContenu(cached.content);
                    
                    // 3. Vérifier les mises à jour en arrière-plan (si connecté)
                    if (navigator.onLine) {
                        setTimeout(() => miseAJourSilencieuse(), 3000);
                    }
                } else {
                    // Pas de cache - télécharger
                    console.log("📭 Aucun cache, téléchargement...");
                    if (progressText) progressText.innerText = 'Premier chargement...';
                    
                    if (!navigator.onLine) {
                        throw new Error("Pas de connexion internet. Connectez-vous pour le premier chargement.");
                    }
                    
                    await telechargerEtAfficher(true);
                }
                
                // 4. Service Worker
                setTimeout(() => registerServiceWorker(), 1000);
                
                // 5. Écouter les changements de connexion
                window.addEventListener('online', () => {
                    console.log("🟢 Connexion restaurée");
                    if (currentContent) {
                        setTimeout(() => miseAJourSilencieuse(), 2000);
                    }
                });
                
            } catch (error) {
                console.error("Erreur fatale:", error);
                afficherErreur(error.message);
            }
        }
